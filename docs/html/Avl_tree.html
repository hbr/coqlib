Library Avl_tree

Require Import Core.
Set Implicit Arguments.

Module Make (S:SORTABLE).
  Import Relation.

  Section sortable.
    Definition A:Set := S.T.

    Theorem transitive: Transitive S.Less_equal.

    Theorem reflexive: Reflexive S.Less_equal.
  End sortable.
  Notation "a <= b" := (S.Less_equal a b) (at level 70).

  Section basic_definitions.
    Inductive Balance: Set :=
      balanced: Balance
    | left_leaning: Balance.

    Record Characteristics: Set :=
      make_characteristics {
          height: nat;
          balance: Balance;
          low_guard: A;
          high_guard: A}.

    Inductive T :=
      empty: T
    | node: Balance → A → T → T → T.

    Inductive Avl_height: nat → Balance → T → Prop :=
      height_empty:
        Avl_height 0 balanced empty
    | height_balanced:
        ∀ h b1 t1 b2 t2 a,
          Avl_height h b1 t1 →
          Avl_height h b2 t2 →
          Avl_height (1+h) balanced (node balanced a t1 t2)
    | height_left_leaning:
        ∀ h b1 t1 b2 t2 a,
          Avl_height (1+h) b1 t1 →
          Avl_height h b2 t2 →
          Avl_height (2+h) left_leaning (node left_leaning a t1 t2).

    Inductive Avl_sort: A → A → T → Prop :=
      sort_empty:
        ∀ lo hi, Avl_sort lo hi empty
    | sort_node:
        ∀ lo1 hi1 t1 lo2 hi2 t2 b a,
          Avl_sort lo1 hi1 t1 →
          Avl_sort lo2 hi2 t2 →
          hi1 ≤ a →
          a ≤ lo2 →
          Avl_sort lo1 hi2 (node b a t1 t2).

    Definition Avl (t:T): Prop
      := ∃ c:Characteristics,
        Avl_height c.(height) (balance c) t ∧
        Avl_sort (low_guard c) (high_guard c) t.
  End basic_definitions.

  Section draft.
    Inductive Avl2: A → A → nat → Balance → T → Prop :=
      avl_empty:
        ∀ lo hi, lo ≤ hi → Avl2 lo hi 0 balanced empty
    | avl_balanced:
        ∀ lo1 hi1 b1 t1 lo2 hi2 b2 t2 h a,
          Avl2 lo1 hi1 h b1 t1 →
          Avl2 lo2 hi2 h b2 t2 →
          hi1 ≤ a →
          a ≤ lo2 →
          Avl2 lo1 hi2 (1+h) balanced (node balanced a t1 t2)
    | avl_left_leaning:
        ∀ lo1 hi1 b1 t1 lo2 hi2 b2 t2 h a,
          Avl2 lo1 hi1 (1+h) b1 t1 →
          Avl2 lo2 hi2 h b2 t2 →
          hi1 ≤ a →
          a ≤ lo2 →
          Avl2 lo1 hi2 (2+h) left_leaning (node left_leaning a t1 t2).

    Theorem extend_guards:
      ∀ (lo_new lo hi hi_new:A) (h:nat) (b:Balance) (t:T),
        Avl2 lo hi h b t →
        lo_new ≤ lo →
        hi ≤ hi_new →
        Avl2 lo_new hi_new h b t.
  End draft.
End Make.
